using Microsoft.Extensions.Logging;
using SalesforcAPI.Interfaces;
using SalesforcAPI.Models;
using System.Globalization;
using System.Xml;
using System.Xml.Linq;

namespace SalesforcAPI.Repositories
{
    public class AnticipatedEndowmentRepository : IAnticipatedEndowmentRepository
    {
        private readonly IRequestSoap _requestSoap;
        private readonly ILogger<AnticipatedEndowmentRepository> _logger;

        public AnticipatedEndowmentRepository(
            IRequestSoap requestSoap,
            ILogger<AnticipatedEndowmentRepository> logger)
        {
            _requestSoap = requestSoap;
            _logger = logger;
        }

        public async Task<AnticipatedEndowmentResponse> GetAnticipatedEndowmentAsync(string policyId)
        {
            if (string.IsNullOrWhiteSpace(policyId))
                throw new ArgumentException("PolicyID tidak boleh kosong.", nameof(policyId));

            string url = "http://10.154.5.45:8093/TXLife/services/TXLife";

            var soapNs = XNamespace.Get("http://schemas.xmlsoap.org/soap/envelope/");
            var txNs = XNamespace.Get("http://schema/webservices.elink.solcorp.com");
            var acordNs = XNamespace.Get("http://ACORD.org/Standards/Life/2");
            var xsiNs = XNamespace.Get("http://www.w3.org/2001/XMLSchema-instance");
            var xsdNs = XNamespace.Get("http://www.w3.org/2001/XMLSchema");
            var schemaLocation = "http://ACORD.org/Standards/Life/2../xsd/TXLife2.9.00.xsd";

            string responseState = string.Empty;
            string responseStatec = string.Empty;
            string responseStated = string.Empty;

            try
            {
                var xmlEndw = new XElement(soapNs + "Envelope",
                    new XAttribute(XNamespace.Xmlns + "soapenv", soapNs),
                    new XAttribute(XNamespace.Xmlns + "web", txNs),
                    new XElement(soapNs + "Header"),
                    new XElement(soapNs + "Body",
                        new XElement(acordNs + "TXLife",
                            new XAttribute(XNamespace.Xmlns + "xsi", xsiNs),
                            new XAttribute(XNamespace.Xmlns + "xsd", xsdNs),
                            new XAttribute(xsiNs + "schemaLocation", schemaLocation),
                            new XElement(acordNs + "SignOnCompany", "CP"),
                            new XElement(acordNs + "UserAuthRequest",
                                new XElement(acordNs + "UserLoginName", "fi80w"),
                                new XElement(acordNs + "UserPswd",
                                    new XElement(acordNs + "CryptType", "NONE"),
                                    new XElement(acordNs + "Pswd", "fi80w")),
                                new XElement(acordNs + "UserCoId", "CP")),
                            new XElement(acordNs + "TXLifeRequest",
                                new XElement(acordNs + "TransRefGUID", ""),
                                new XElement(acordNs + "TransType", new XAttribute("tc", "InquiryAnticipatedEndowment")),
                                new XElement(acordNs + "TransExeDate", ""),
                                new XElement(acordNs + "TransExeTime", ""),
                                new XElement(acordNs + "OLifE",
                                    new XElement(acordNs + "InquiryAnticipatedEndowmentData",
                                        new XElement(acordNs + "MirCommonFields",
                                            new XElement(acordNs + "MirDvEffDt", "16APR2026"),
                                            new XElement(acordNs + "MirCvgNum", "01"),
                                            new XElement(acordNs + "MirPolPayoNum", "00001")),
                                        new XElement(acordNs + "MirPolId",
                                            new XElement(acordNs + "MirPolIdBase", policyId),
                                            new XElement(acordNs + "MirPolIdSfx", ""))))))));

                responseState = await _requestSoap.PostSOAPRequestAsync(url, xmlEndw.ToString());

                var xmlCallCentre = new XElement(soapNs + "Envelope",
                    new XAttribute(XNamespace.Xmlns + "soapenv", soapNs),
                    new XAttribute(XNamespace.Xmlns + "web", txNs),
                    new XElement(soapNs + "Header"),
                    new XElement(soapNs + "Body",
                        new XElement(acordNs + "TXLife",
                            new XAttribute(XNamespace.Xmlns + "xsi", xsiNs),
                            new XAttribute(XNamespace.Xmlns + "xsd", xsdNs),
                            new XAttribute(xsiNs + "schemaLocation", schemaLocation),
                            new XElement(acordNs + "SignOnCompany", "CP"),
                            new XElement(acordNs + "UserAuthRequest",
                                new XElement(acordNs + "UserLoginName", "fi80w"),
                                new XElement(acordNs + "UserPswd",
                                    new XElement(acordNs + "CryptType", "NONE"),
                                    new XElement(acordNs + "Pswd", "fi80w")),
                                new XElement(acordNs + "UserCoId", "CP")),
                            new XElement(acordNs + "TXLifeRequest",
                                new XElement(acordNs + "TransRefGUID", ""),
                                new XElement(acordNs + "TransType", new XAttribute("tc", "CallCentreInquiry")),
                                new XElement(acordNs + "TransExeDate", ""),
                                new XElement(acordNs + "TransExeTime", ""),
                                new XElement(acordNs + "OLifE",
                                    new XElement(acordNs + "CallCentreInquiryData",
                                        new XElement(acordNs + "MirPolId",
                                            new XElement(acordNs + "MirPolIdBase", policyId),
                                            new XElement(acordNs + "MirPolIdSfx", "")),
                                        new XElement(acordNs + "MirCvgNum", ""),
                                        new XElement(acordNs + "MirDvEffDt", ""))))))));

                responseStatec = await _requestSoap.PostSOAPRequestAsync(url, xmlCallCentre.ToString());

                var xmlContractualPayout = new XElement(soapNs + "Envelope",
                    new XAttribute(XNamespace.Xmlns + "soapenv", soapNs),
                    new XAttribute(XNamespace.Xmlns + "web", txNs),
                    new XElement(soapNs + "Header"),
                    new XElement(soapNs + "Body",
                        new XElement(acordNs + "TXLife",
                            new XAttribute(XNamespace.Xmlns + "xsi", xsiNs),
                            new XAttribute(XNamespace.Xmlns + "xsd", xsdNs),
                            new XAttribute(xsiNs + "schemaLocation", schemaLocation),
                            new XElement(acordNs + "SignOnCompany", "CP"),
                            new XElement(acordNs + "UserAuthRequest",
                                new XElement(acordNs + "UserLoginName", "fi80w"),
                                new XElement(acordNs + "UserPswd",
                                    new XElement(acordNs + "CryptType", "NONE"),
                                    new XElement(acordNs + "Pswd", "fi80w")),
                                new XElement(acordNs + "UserCoId", "CP")),
                            new XElement(acordNs + "TXLifeRequest",
                                new XElement(acordNs + "TransRefGUID", ""),
                                new XElement(acordNs + "TransType", new XAttribute("tc", "InquiryContractualPayouts")),
                                new XElement(acordNs + "TransExeDate", ""),
                                new XElement(acordNs + "TransExeTime", ""),
                                new XElement(acordNs + "OLifE",
                                    new XElement(acordNs + "InquiryContractualPayoutsData",
                                        new XElement(acordNs + "MirCommonFields",
                                            new XElement(acordNs + "MirCvgNum", ""),
                                            new XElement(acordNs + "MirDvEffDt", ""),
                                            new XElement(acordNs + "MirPolId",
                                                new XElement(acordNs + "MirPolIdBase", policyId),
                                                new XElement(acordNs + "MirPolIdSfx", ""))))))))));

                responseStated = await _requestSoap.PostSOAPRequestAsync(url, xmlContractualPayout.ToString());

                _logger.LogInformation(
                    "[AnticipatedEndowment] SOAP success PolicyID={PolicyID}, EndowmentLength={EndowmentLength}, CallCentreLength={CallCentreLength}, ContractualPayoutsLength={ContractualPayoutsLength}",
                    policyId,
                    responseState?.Length ?? 0,
                    responseStatec?.Length ?? 0,
                    responseStated?.Length ?? 0);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "[AnticipatedEndowment] SOAP request gagal PolicyID={PolicyID}", policyId);
                throw;
            }

            try
            {
                var payoutDates = new List<PayoutDateInformation>();
                var contractualInfo = new ContractualPayoutInformation { PolicyID = policyId };
                var aeFundInfo = new AEFundInformation { PolicyID = policyId };

                XmlNamespaceManager CreateNs(XmlDocument doc)
                {
                    var ns = new XmlNamespaceManager(doc.NameTable);
                    ns.AddNamespace("soap", "http://schemas.xmlsoap.org/soap/envelope/");
                    ns.AddNamespace("ns", "http://ACORD.org/Standards/Life/2");
                    return ns;
                }

                // VALIDASI POLICY ADA / TIDAK
                if (string.IsNullOrWhiteSpace(responseState))
                    throw new KeyNotFoundException($"PolicyID {policyId} tidak ditemukan.");

                var checkDoc = new XmlDocument();
                checkDoc.LoadXml(XDocument.Parse(responseState).ToString());
                var checkNs = CreateNs(checkDoc);

                var endowmentNode = checkDoc.SelectSingleNode("//ns:InquiryAnticipatedEndowmentData", checkNs);
                var policyNode = checkDoc.SelectSingleNode("//ns:InquiryAnticipatedEndowmentData/ns:MirPolId/ns:MirPolIdBase", checkNs);

                bool policyNotFound =
                    endowmentNode == null ||
                    policyNode == null ||
                    string.IsNullOrWhiteSpace(policyNode.InnerText);

                if (policyNotFound)
                    throw new KeyNotFoundException($"PolicyID {policyId} tidak ditemukan.");

                if (!string.IsNullOrWhiteSpace(responseStated))
                {
                    var xmlDoc = new XmlDocument();
                    xmlDoc.LoadXml(XDocument.Parse(responseStated).ToString());

                    var ns = CreateNs(xmlDoc);
                    var nodes = xmlDoc.SelectNodes("//ns:InquiryContractualPayoutsData", ns);
                    int seq = 1;

                    if (nodes != null)
                    {
                        foreach (XmlNode node in nodes)
                        {
                            string payoutDateText = node.SelectSingleNode("ns:MirDvXpctPayoDtG/ns:MirDvXpctPayoDtT", ns)?.InnerText ?? "";
                            string payoutAmountText = node.SelectSingleNode("ns:MirDvXpctPayoAmtG/ns:MirDvXpctPayoAmtT", ns)?.InnerText ?? "";
                            string payoutStatusText = node.SelectSingleNode("ns:MirCdiPayomthdCdG/ns:MirCdiPayomthdCdT", ns)?.InnerText ?? "";

                            if (string.IsNullOrWhiteSpace(payoutDateText) &&
                                string.IsNullOrWhiteSpace(payoutAmountText) &&
                                string.IsNullOrWhiteSpace(payoutStatusText))
                                continue;

                            payoutDates.Add(new PayoutDateInformation
                            {
                                PolicyID = policyId,
                                PayoutDate = ParseDate(payoutDateText),
                                PayoutAmount = ParseDouble(payoutAmountText),
                                PayoutStatus = payoutStatusText,
                                SequenceNumber = seq++
                            });
                        }
                    }
                }

                {
                    var xmlDoc = new XmlDocument();
                    xmlDoc.LoadXml(XDocument.Parse(responseState).ToString());

                    var ns = CreateNs(xmlDoc);
                    var node = xmlDoc.SelectSingleNode("//ns:InquiryAnticipatedEndowmentData", ns)!;

                    contractualInfo = new ContractualPayoutInformation
                    {
                        PolicyID = policyId,
                        PayoutMethod = node.SelectSingleNode("ns:MirCdiPayomthdCdG/ns:MirCdiPayomthdCdT", ns)?.InnerText ?? "",
                        PayoutNumber = ParseInt(node.SelectSingleNode("ns:MirCommonFields/ns:MirPolPayoNum", ns)?.InnerText),
                        NextPayoutDate = ParseDate(node.SelectSingleNode("ns:MirCommonFields/ns:MirPolPayoNxtDt", ns)?.InnerText),
                        LastPayoutDate = ParseDate(node.SelectSingleNode("ns:MirCommonFields/ns:MirCdaEffDt", ns)?.InnerText),
                        LastPayoutAmount = ParseDouble(node.SelectSingleNode("ns:MirCommonFields/ns:MirLastPayoAmt", ns)?.InnerText),
                        TotalPayoutAmount = ParseDouble(node.SelectSingleNode("ns:MirCommonFields/ns:MirPolPayoAmt", ns)?.InnerText),
                        IssueDate = ParseDate(node.SelectSingleNode("ns:MirCommonFields/ns:MirPolPayoEffDt", ns)?.InnerText)
                    };

                    aeFundInfo = new AEFundInformation
                    {
                        PolicyID = policyId,
                        AEFund = node.SelectSingleNode("ns:MirValuotrFields/ns:MirPolAeAmt", ns)?.InnerText ?? "",
                        AEFundInterest = node.SelectSingleNode("ns:MirValuotrFields/ns:MirPolAeIntAmt", ns)?.InnerText ?? "",
                        LastWithdrawalDate = ParseDate(node.SelectSingleNode("ns:MirCommonFields/ns:MirLastAeWthdrDt", ns)?.InnerText),
                        LoanBalance = ParseDouble(node.SelectSingleNode("ns:MirValuInFields/ns:MirDvLoanRepayAmt", ns)?.InnerText),
                        PlanName = node.SelectSingleNode("ns:MirCommonFields/ns:MirDvFndDescTxt", ns)?.InnerText ?? "",
                        LastWithdrawalAmount = ParseDouble(node.SelectSingleNode("ns:MirCommonFields/ns:MirLastAeWthdrAmt", ns)?.InnerText)
                    };
                }

                if (payoutDates.Count == 0)
                {
                    payoutDates.Add(new PayoutDateInformation
                    {
                        PolicyID = policyId,
                        PayoutDate = null,
                        PayoutAmount = 0,
                        PayoutStatus = "",
                        SequenceNumber = 0
                    });
                }

                return new AnticipatedEndowmentResponse
                {
                    ContractualPayoutInfo = contractualInfo,
                    PayoutDates = payoutDates,
                    AEFundInfo = aeFundInfo
                };
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "[AnticipatedEndowment] Parse XML gagal PolicyID={PolicyID}", policyId);
                throw;
            }
        }

        private static DateTime? ParseDate(string? value)
        {
            if (string.IsNullOrWhiteSpace(value)) return null;

            string[] formats = { "ddMMyyyy", "dd/MM/yyyy", "yyyy-MM-dd", "ddMMMyyyy" };
            if (DateTime.TryParseExact(value, formats, CultureInfo.InvariantCulture, DateTimeStyles.None, out var dt))
                return dt;

            return DateTime.TryParse(value, out var fallback) ? fallback : null;
        }

        private static double ParseDouble(string? value)
        {
            if (string.IsNullOrWhiteSpace(value)) return 0;
            return double.TryParse(value, NumberStyles.Any, CultureInfo.InvariantCulture, out var d) ? d : 0;
        }

        private static int ParseInt(string? value)
        {
            if (string.IsNullOrWhiteSpace(value)) return 0;
            return int.TryParse(value, out var i) ? i : 0;
        }
    }
}

using Microsoft.AspNetCore.Mvc;
using SalesforcAPI.Interfaces;
using SalesforcAPI.Models;
using System.Xml;

namespace SalesforcAPI.Controllers
{
    [Route("api/[controller]")]
    [ApiController]
    public class AnticipatedEndowmentController : ControllerBase
    {
        private readonly IAnticipatedEndowmentRepository _repository;
        private readonly ILogger<AnticipatedEndowmentController> _logger;

        public AnticipatedEndowmentController(
            IAnticipatedEndowmentRepository repository,
            ILogger<AnticipatedEndowmentController> logger)
        {
            _repository = repository;
            _logger = logger;
        }

        [HttpPost]
        [ProducesResponseType(typeof(AnticipatedEndowmentResponse), StatusCodes.Status200OK)]
        [ProducesResponseType(typeof(ApiErrorResponse), StatusCodes.Status400BadRequest)]
        [ProducesResponseType(typeof(ApiErrorResponse), StatusCodes.Status500InternalServerError)]
        public async Task<ActionResult<AnticipatedEndowmentResponse>> SendSoapRequestAsync(
            [FromBody] AnticipatedEndowmentRequest request)
        {
            if (!ModelState.IsValid)
            {
                var detail = string.Join("; ", ModelState.Values
                    .SelectMany(v => v.Errors)
                    .Select(e => e.ErrorMessage));

                return BadRequest(new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status400BadRequest,
                    Message = "Request tidak valid.",
                    Detail = detail,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }

            try
            {
                var response = await _repository.GetAnticipatedEndowmentAsync(request.PolicyID)
                               ?? BuildEmptyResponse(request.PolicyID);

                return new JsonResult(response)
                {
                    StatusCode = StatusCodes.Status200OK
                };
            }
            catch (ArgumentException ex)
            {
                _logger.LogWarning(ex,
                    "[AnticipatedEndowment] Parameter tidak valid | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status400BadRequest, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status400BadRequest,
                    Message = "Parameter tidak valid.",
                    Detail = ex.Message,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
            catch (KeyNotFoundException ex)
            {
                _logger.LogWarning(ex,
                    "[AnticipatedEndowment] Policy tidak ditemukan | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status400BadRequest, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status400BadRequest,
                    Message = "Policy tidak ditemukan.",
                    Detail = ex.Message,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
            catch (XmlException ex)
            {
                _logger.LogError(ex,
                    "[AnticipatedEndowment] XML parse error | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status500InternalServerError, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status500InternalServerError,
                    Message = "Gagal memproses response XML dari SOAP service.",
                    Detail = ex.Message,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
            catch (HttpRequestException ex)
            {
                _logger.LogError(ex,
                    "[AnticipatedEndowment] HTTP error | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status500InternalServerError, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status500InternalServerError,
                    Message = "Service Unavailable.",
                    Detail = ex.Message,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
            catch (TaskCanceledException ex)
            {
                _logger.LogError(ex,
                    "[AnticipatedEndowment] Timeout | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status500InternalServerError, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status500InternalServerError,
                    Message = "Request Timeout.",
                    Detail = "Koneksi ke TXLife melebihi batas waktu.",
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
            catch (Exception ex)
            {
                _logger.LogError(ex,
                    "[AnticipatedEndowment] Error tidak terduga | PolicyID: {PolicyID}",
                    request.PolicyID);

                return StatusCode(StatusCodes.Status500InternalServerError, new ApiErrorResponse
                {
                    Success = false,
                    StatusCode = StatusCodes.Status500InternalServerError,
                    Message = "Terjadi kesalahan internal.",
                    Detail = ex.Message,
                    Data = BuildEmptyResponse(request.PolicyID)
                });
            }
        }

        private static AnticipatedEndowmentResponse BuildEmptyResponse(string policyId) =>
            new AnticipatedEndowmentResponse
            {
                ContractualPayoutInfo = new ContractualPayoutInformation
                {
                    PolicyID = policyId,
                    PayoutMethod = "",
                    PayoutNumber = 0,
                    NextPayoutDate = null,
                    LastPayoutDate = null,
                    LastPayoutAmount = 0,
                    TotalPayoutAmount = 0,
                    IssueDate = new DateTime(1900, 1, 1)
                },
                AEFundInfo = new AEFundInformation
                {
                    PolicyID = policyId,
                    AEFund = "",
                    AEFundInterest = "",
                    LoanBalance = 0,
                    PlanName = "",
                    LastWithdrawalAmount = 0,
                    LastWithdrawalDate = null
                },
                PayoutDates = new List<PayoutDateInformation>
                {
                    new PayoutDateInformation
                    {
                        PolicyID = policyId,
                        PayoutDate = null,
                        PayoutAmount = 0,
                        PayoutStatus = "",
                        SequenceNumber = 0
                    }
                }
            };
    }
